    function adjustColor(color, percent) {

      const num = parseInt(color.replace("#", ""), 16);

      const amt = Math.round(2.55 * percent);

      const R = (num >> 16) + amt;

      const G = (num >> 8 & 0x00FF) + amt;

      const B = (num & 0x0000FF) + amt;

      return "#" + (0x1000000 + (R < 255 ? R < 1 ? 0 : R : 255) * 0x10000 +

        (G < 255 ? G < 1 ? 0 : G : 255) * 0x100 +

        (B < 255 ? B < 1 ? 0 : B : 255))

        .toString(16).slice(1);

    }

    async function incrementCounter() {

      const button = document.getElementById('counter-button');

      const display = document.getElementById('counter-display');

     

      button.disabled = true;

      button.classList.add('loading');

     

      localPrayerCount++;

      display.textContent = `__ _________ __ ${localPrayerCount} ___`;

     

      const result = await window.dataSdk.create({

        prayer_count: 1,

        timestamp: new Date().toISOString()

      });

     

      button.disabled = false;

      button.classList.remove('loading');

     

      if (!result.isOk) {

        localPrayerCount--;

        display.textContent = `__ _________ __ ${localPrayerCount} ___`;

       

        const errorMsg = document.createElement('p');

        errorMsg.textContent = '___ ____ ____ ________ ___ ____';

        errorMsg.style.color = '#EF4444';

        errorMsg.style.marginTop = '1rem';

        errorMsg.id = 'error-message';

       

        const existingError = document.getElementById('error-message');

        if (existingError) {

          existingError.remove();

        }

       

        document.querySelector('.counter-container').appendChild(errorMsg);

       

        setTimeout(() => {

          errorMsg.remove();

        }, 3000);

      }

    }

 

    document.getElementById('counter-button').addEventListener('click', incrementCounter);
